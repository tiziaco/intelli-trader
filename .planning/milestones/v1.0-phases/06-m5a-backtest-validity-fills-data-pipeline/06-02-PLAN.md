class PriceStore(ABC):
    @abstractmethod
    def read_bars(self, ticker: str) -> pd.DataFrame: ...
        # full canonical frame (tz-aware index named 'date', float64 columns
        # open/high/low/close/volume); raises MissingPriceDataError for unknown ticker
    @abstractmethod
    def write_bars(self, ticker: str, frame: pd.DataFrame) -> None: ...
        # offline ingestion surface; CsvPriceStore raises NotImplementedError (read-only run path)
    @abstractmethod
    def has(self, ticker: str) -> bool: ...
    @abstractmethod
    def symbols(self) -> list[str]: ...
    @abstractmethod
    def index(self, ticker: str) -> pd.DatetimeIndex: ...
        # feeds TimeGenerator.set_dates; raises MissingPriceDataError for unknown ticker

class CsvPriceStore(PriceStore):
    CSV_DEFAULT_PATH = 'data/BTCUSD_1d_ohlcv_2018_2026.csv'
    CSV_START_DATE = '2018-01-01'
    CSV_END_DATE = '2026-06-03'
    CSV_TICKER = 'BTCUSD'
    def __init__(self, csv_paths: dict[str, str | Path] | None = None,
                 start_date: str | None = None, end_date: str | None = None) -> None: ...
        # None -> {CSV_TICKER: CSV_DEFAULT_PATH}; loads eagerly at construction;
        # multi-symbol mapping support enables the 06-03 megaframe fixture

class PriceProvider(ABC):  # providers/base.py — offline only
    @abstractmethod
    def fetch_ohlcv(self, symbol: str, timeframe: str, start: str,
                    end: str | None = None) -> pd.DataFrame: ...
    @abstractmethod
    def get_symbols(self) -> list[str]: ...
