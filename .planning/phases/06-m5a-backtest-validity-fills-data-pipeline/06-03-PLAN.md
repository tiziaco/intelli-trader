```python
class BarFeed(ABC):
    @abstractmethod
    def current_bars(self, time: datetime) -> dict[str, Bar]: ...
        # one Bar per ticker that has a base bar stamped exactly `time`;
        # tickers with no bar at `time` are ABSENT (sparse universe, D-15)
    @abstractmethod
    def window(self, ticker: str, timeframe: timedelta, max_window: int,
               asof: datetime) -> pd.DataFrame: ...
        # float64 frame (D-17), tz-aware DatetimeIndex, completed bars only;
        # raises MissingPriceDataError for unknown ticker (FR7)
    @abstractmethod
    def megaframe(self, asof: datetime, timeframe: timedelta,
                  max_window: int) -> pd.DataFrame: ...
        # D-19: multi-symbol concat, keys = actually-included symbols, tz-aware

class BacktestBarFeed(BarFeed):  # feed/bar_feed.py
    def __init__(self, store: PriceStore, base_timeframe: timedelta) -> None: ...
    def precompute(self, tickers: list[str], timeframe: timedelta) -> None: ...
        # eager resample at run-init from strategy declarations (06-05 calls this);
        # undeclared timeframes lazily compute-and-memoize on first access
```
